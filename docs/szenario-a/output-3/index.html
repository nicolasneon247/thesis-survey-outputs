    internal procedure UpdateMarginAndSalesPrice()
    var
        MarginSetup: Record "TSRD Margin Setup";
        CalcForecastMgt: Codeunit "TSRD Calc. Forecast Mgt.";
        MinimumMargin: Decimal;
    begin
        SalesReceivablesSetup.Get();
        if "Sales Price" = 0 then
            exit;
        if "Includes Sales Discount" then begin
            "Sales Price" := GetCalculatedUnitPrice();
            UpdateSalesPriceSCC();
            "Margin Amount" := GetCalculatedUnitPrice() - GetCalculatedUnitCost();
            "Margin %" := CalcForecastMgt.GetMarginPercentageFromSalesPrice(SalesReceivablesSetup."TSRD Margin Model", GetCalculatedUnitCost(), GetCalculatedUnitPrice());
            MinimumMargin := MarginSetup.GetMinimumMarginPercentage(Rec."Manufacturer Code", Rec."Category Code",
                                                               Rec."Subcategory Code", Rec."Series Code", Rec."Model Code");
            if "Margin %" < MinimumMargin then begin
                if CurrFieldNo = FieldNo("Base Machine List Price") then
                    "Base Machine List Price" := xRec."Base Machine List Price";
                UpdateListPrice();
                FixAccordingtoMinimumMargin();
            end;
        end
        else begin
            MinimumMargin := MarginSetup.GetMinimumMarginPercentage(Rec."Manufacturer Code", Rec."Category Code",
                                                                Rec."Subcategory Code", Rec."Series Code", Rec."Model Code");
            if "Margin %" < MinimumMargin then
                FixAccordingtoMinimumMargin()
            else begin
                "Margin %" := CalcForecastMgt.GetMarginPercentageFromSalesPrice(SalesReceivablesSetup."TSRD Margin Model", GetCalculatedUnitCost(), "Sales Price");
                if "Margin %" < MinimumMargin then
                    FixAccordingtoMinimumMargin();
                "Margin Amount" := "Sales Price" - GetCalculatedUnitCost();
            end;
        end;
        Rec.Modify();
    end;

    local procedure GetCalculatedMachineSalesPrice(var PriceWithoutMargin: Decimal; var SalesPriceWithMargin: Decimal; var PurchDiscountMultiplier: Decimal; Currency: Record Currency)
    var
        CalcForecastMgt: Codeunit "TSRD Calc. Forecast Mgt.";
    begin
        CalcFields("Total Discount", "Dir unit cost with Calc. Disc", "Configuration Price", "List Price Conf. Options");
        if "Show List Prices" then
            PriceWithoutMargin := "List Price Conf. Options" + GetBaseMachineListPriceForSalesPriceCalc() - "Total Discount"
        else
            PriceWithoutMargin := "Configuration Price" + GetBaseMachineListPriceForSalesPriceCalc() - "Total Discount";

        SalesPriceWithMargin := Round(CalcForecastMgt.GetSalesPriceWithMargin(PriceWithoutMargin,
                                                                              "Margin %",
                                                                              SalesReceivablesSetup."TSRD Margin Model"), Currency."Amount Rounding Precision");
        if (SalesPriceWithMargin = 0) or ("List Price" = 0) then begin
            "Calculated Machine Sales Price" := 0;
            exit;
        end;
        if "Total Discount" <> 0 then
            PurchDiscountMultiplier := 1 - "Total Discount" / ("Dir unit cost with Calc. Disc" + GetBaseMachineListPriceForSalesPriceCalc())
        else
            PurchDiscountMultiplier := 1;

        "Calculated Machine Sales Price" := Round(CalcForecastMgt.GetSalesPriceWithMargin(GetBaseMachineListPriceForSalesPriceCalc() * PurchDiscountMultiplier, "Margin %", SalesReceivablesSetup."TSRD Margin Model"), Currency."Amount Rounding Precision");
    end;

    local procedure UpdateCalcConfigLinesCalculatedSalesPrices(var CalcConfiguration: Record "TSRD Calc. Configuration"; var SalesPriceSplitTotal: Decimal; PurchDiscountMultiplier: Decimal; Currency: Record Currency)
    var
        CalcForecastMgt: Codeunit "TSRD Calc. Forecast Mgt.";
    begin
        CalcConfiguration.SetRange("Calculate Discount", true);
        if not CalcConfiguration.IsEmpty() then begin
            CalcConfiguration.FindSet();
            repeat
                CalcConfiguration."Calculated Sales Amount" := Round(CalcForecastMgt.GetSalesPriceWithMargin(CalcConfiguration."Line Amount (Direct Unit Cost)" * PurchDiscountMultiplier, "Margin %", SalesReceivablesSetup."TSRD Margin Model"), Currency."Amount Rounding Precision");

                CalcConfiguration.Modify();
                SalesPriceSplitTotal += CalcConfiguration."Calculated Sales Amount";
            until CalcConfiguration.Next() = 0;
        end;

        CalcConfiguration.SetRange("Calculate Discount", false);
        if not CalcConfiguration.IsEmpty() then begin
            CalcConfiguration.FindSet();
            repeat
                CalcConfiguration."Calculated Sales Amount" := Round(CalcForecastMgt.GetSalesPriceWithMargin(CalcConfiguration."Line Amount (Direct Unit Cost)", "Margin %", SalesReceivablesSetup."TSRD Margin Model"), Currency."Amount Rounding Precision");
                CalcConfiguration."Calculated Sales Price" := CalcConfiguration."Calculated Sales Amount" / CalcConfiguration.Quantity;
                CalcConfiguration.Modify();
                SalesPriceSplitTotal += CalcConfiguration."Calculated Sales Amount";
            until CalcConfiguration.Next() = 0;
        end;
    end;

    local procedure CalcUnitPrices()
    begin
        CalculatedUnitCosts := Rec.GetCalculatedUnitCost();
        if Rec."Includes Sales Discount" then
            SalesPriceWithoutSeparately := Rec.GetCalculatedUnitPrice()
        else
            SalesPriceWithoutSeparately := Rec.GetCalculatedUnitCost();
        Rec.CalcFields("Equipment (Separate)", "Workshop (Separate)", "Over-Allowance Price",
                       "Additional Config. (Sep.) SP", "Additional Config. (Sep.) Cost",
                       "Services (Separate) SP", "Services (Separate) Amount",
                       "Costs (Separate) SP", "Costs (Separate) Amount");
        if Rec."Over-Allowance Price" <> 0 then begin
            ShowOverAllowancePrice := true;
            PurchaseTradeInAmt := CalcForecastMgt.CalculatePurchaseAmount(Rec, true) + Rec."Over-Allowance Price";
        end else begin
            ShowOverAllowancePrice := false;
            PurchaseTradeInAmt := CalcForecastMgt.CalculatePurchaseAmount(Rec, false);
        end;

        SalesPriceTotal := Rec."Sales Price"
                           + Rec."Additional Config. (Sep.) SP"
                           + Rec."Services (Separate) SP"
                           + Rec."Costs (Separate) SP";
        TotalCostsAmount := CalculatedUnitCosts
                            + Rec."Additional Config. (Sep.) Cost"
                            + Rec."Services (Separate) Amount"
                            + Rec."Costs (Separate) Amount";
        SalesReceivablesSetup.Get();
        TotalMargin := SalesPriceTotal - TotalCostsAmount;
