codeunit 89900 "DMS29748 Test Setup"
{
    // Shared deterministic setup for feature 29748 / work items 27490, 31605, 31606, 31727.
    Access = Public;

    var
        LibrarySales: Codeunit "Library - Sales";

    procedure CreateCustomerAndSalesForecast(var CalcSalesForecast: Record "TSRD Calc. Sales Forecast"): Boolean
    var
        Customer: Record Customer;
    begin
        LibrarySales.CreateCustomer(Customer);
        CalcSalesForecast.Init();
        if not CalcSalesForecast.Insert(true) then
            exit(false);
        CalcSalesForecast.Validate("Customer No.", Customer."No.");
        CalcSalesForecast.Modify(true);
        exit(true);
    end;

    procedure TryInsertCalcForecastFromFirstServiceItem(var CalcSalesForecast: Record "TSRD Calc. Sales Forecast"; var CalcForecast: Record "TSRD Calc. Forecast"): Boolean
    var
        ServiceItem: Record "Service Item";
        Vendor: Record Vendor;
        CalcForecastMgt: Codeunit "TSRD Calc. Forecast Mgt.";
    begin
        if not ServiceItem.FindFirst() then
            exit(false);

        if ServiceItem."Vendor No." <> '' then
            Vendor.Get(ServiceItem."Vendor No.")
        else
            if not Vendor.FindFirst() then
                exit(false);

        CalcForecastMgt.InsertForecast(ServiceItem, Vendor, CalcSalesForecast, CalcForecast);
        exit(true);
    end;

    procedure EnsureTestCostType(var CalcCostType: Record "TSRD Calc. Cost Type"; ItemChargeNo: Code[20]): Boolean
    begin
        if CalcCostType.Get('DMS29748-CT') then
            exit(true);
        if ItemChargeNo = '' then
            exit(false);
        CalcCostType.Init();
        CalcCostType.Code := 'DMS29748-CT';
        CalcCostType.Description := 'DMS 29748 test cost type';
        CalcCostType.Validate("Item Charge", ItemChargeNo);
        CalcCostType.Validate("Sales Item Charge", ItemChargeNo);
        exit(CalcCostType.Insert(true));
    end;

    /// <summary>Uses the first Item Charge in the company; cost type validate must accept it for both purchase and sales (e.g. Cost of Sale in both relations).</summary>
    procedure FindItemChargeForCalcCostType(var ItemChargeNo: Code[20]): Boolean
    var
        ItemCharge: Record "Item Charge";
    begin
        if not ItemCharge.FindFirst() then
            exit(false);
        ItemChargeNo := ItemCharge."No.";
        exit(true);
    end;
}
