    procedure EnsureWorkshopResource()
    var
        Resource: Record Resource;
    begin
        if Resource.Get(GetWorkshopResourceNo()) then
            exit;

        Resource.Init();
        Resource."No." := GetWorkshopResourceNo();
        Resource.Name := 'Feature 29748 Workshop';
        Resource.Type := Resource.Type::Person;
        Resource.Insert(true);
    end;

    procedure EnsureSalesItemCharge(ScenarioCode: Code[20]; FillSalesItemCharge: Boolean; var CalcCostType: Record "TSRD Calc. Cost Type")
    var
        ItemCharge: Record "Item Charge";
    begin
        EnsurePurchaseItemCharge(ScenarioCode);

        if not ItemCharge.Get(GetSalesItemChargeNo(ScenarioCode)) then begin
            ItemCharge.Init();
            ItemCharge."No." := GetSalesItemChargeNo(ScenarioCode);
            ItemCharge.Description := 'Feature 29748 Sales Charge';
            ItemCharge.Insert(true);
        end;

        if not CalcCostType.Get(GetCostTypeCode(ScenarioCode)) then begin
            CalcCostType.Init();
            CalcCostType.Code := GetCostTypeCode(ScenarioCode);
            CalcCostType.Description := 'Feature 29748 Cost Type';
            CalcCostType."Item Charge" := GetPurchaseItemChargeNo(ScenarioCode);
            if FillSalesItemCharge then
                CalcCostType."Sales Item Charge" := GetSalesItemChargeNo(ScenarioCode);
            CalcCostType.Insert(true);
            exit;
        end;

        CalcCostType."Item Charge" := GetPurchaseItemChargeNo(ScenarioCode);
        if FillSalesItemCharge then
            CalcCostType."Sales Item Charge" := GetSalesItemChargeNo(ScenarioCode)
        else
            CalcCostType."Sales Item Charge" := '';
        CalcCostType.Modify(true);
    end;

    procedure CreateWorkshopLine(ForecastNo: Code[20]; LineNo: Integer; Description: Text[100]; TotalAmount: Decimal; ChargingMethod: Enum "TSRD Charging Method"; ShowInSalesDocument: Boolean): Record "TSRD Calc. Workshop"
    var
        CalcWorkshop: Record "TSRD Calc. Workshop";
    begin
        CalcWorkshop.Init();

        CalcWorkshop."Line No." := LineNo;

        CalcWorkshop."Amount (Labor)" := TotalAmount;
        CalcWorkshop."Amount (Parts)" := 0;
        CalcWorkshop."Total Amount" := TotalAmount;
        CalcWorkshop."Show in Sales Document" := ShowInSalesDocument;
        CalcWorkshop.Insert(true);
        if ChargingMethod <> CalcWorkshop."Charging Method" then begin
            CalcWorkshop.Validate("Charging Method", ChargingMethod);
            CalcWorkshop.Modify(true);
        end;
        exit(CalcWorkshop);
    end;

    var
        SalesReceivablesSetup: Record "Sales & Receivables Setup";
    begin
        SalesReceivablesSetup.Get();
        EnsureWorkshopResource();
        SalesReceivablesSetup."TSRD Margin Model" := SalesReceivablesSetup."TSRD Margin Model"::Surcharge;
        SalesReceivablesSetup."TSRD Resource No. Workshop" := GetWorkshopResourceNo();

    end;

    local procedure EnsureCustomer(CustomerNo: Code[20])
    var
        Customer: Record Customer;
        CustomerPostingGroup: Record "Customer Posting Group";
        GenBusPostingGroup: Record "Gen. Business Posting Group";
        VATBusPostingGroup: Record "VAT Business Posting Group";
    begin
        if Customer.Get(CustomerNo) then
            exit;

        CustomerPostingGroup.FindFirst();
        GenBusPostingGroup.FindFirst();
        VATBusPostingGroup.FindFirst();

        Customer.Init();
        Customer."No." := CustomerNo;
        Customer.Name := 'Feature 29748 Customer';
        Customer."Customer Posting Group" := CustomerPostingGroup.Code;
        Customer."Gen. Bus. Posting Group" := GenBusPostingGroup.Code;
        Customer."VAT Bus. Posting Group" := VATBusPostingGroup.Code;
        Customer.Insert(true);
    end;

    local procedure EnsureMainItem(ScenarioCode: Code[20])
    var
        Item: Record Item;
        InventoryPostingGroup: Record "Inventory Posting Group";
        GenProdPostingGroup: Record "Gen. Product Posting Group";
        VATProdPostingGroup: Record "VAT Product Posting Group";
    begin
        if Item.Get(GetMainItemNo(ScenarioCode)) then
            exit;

        InventoryPostingGroup.FindFirst();
        GenProdPostingGroup.FindFirst();
        VATProdPostingGroup.FindFirst();

        Item.Init();
        Item."No." := GetMainItemNo(ScenarioCode);
        Item.Description := 'Feature 29748 Item';
        Item.Type := Item.Type::Inventory;
        Item."Base Unit of Measure" := GetBaseUnitOfMeasureCode();
        Item."Inventory Posting Group" := InventoryPostingGroup.Code;
        Item."Gen. Prod. Posting Group" := GenProdPostingGroup.Code;
        Item."VAT Prod. Posting Group" := VATProdPostingGroup.Code;
        Item."Manufacturer Code" := 'TST';
        Item.Insert(true);
    end;

    local procedure EnsurePurchaseItemCharge(ScenarioCode: Code[20])
    var
        ItemCharge: Record "Item Charge";
    begin
        if ItemCharge.Get(GetPurchaseItemChargeNo(ScenarioCode)) then
            exit;

        ItemCharge.Init();
        ItemCharge."No." := GetPurchaseItemChargeNo(ScenarioCode);
        ItemCharge.Description := 'Feature 29748 Purchase Charge';
        ItemCharge.Insert(true);
    end;

    local procedure GetPurchaseItemChargeNo(ScenarioCode: Code[20]): Code[20]
    begin
        exit(CopyStr('PIC-' + ScenarioCode, 1, 20));
    end;

    local procedure GetBaseUnitOfMeasureCode(): Code[10]
    var
        UnitOfMeasure: Record "Unit of Measure";
    begin
        UnitOfMeasure.FindFirst();
        exit(UnitOfMeasure.Code);
    end;
