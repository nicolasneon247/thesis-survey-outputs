/// <summary>
/// Shared setup and test-data factory for feature 29748 tests
/// (Rework price logic for Workshops, Costs and Additional Configuration).
///
/// All helpers insert records directly with deterministic keys and values so
/// tests remain independent from random libraries and from heavy base data.
/// </summary>
codeunit 81501 "TSRD F29748 Charg. Method Lib"
{
    var
        CalcForecastNoTok: Label 'TST-F29748-%1', Locked = true;
        CostTypeWithChargeCodeTok: Label 'TST-CTC-%1', Locked = true;
        CostTypeNoChargeCodeTok: Label 'TST-CTN-%1', Locked = true;
        ItemChargeCodeTok: Label 'TST-IC-%1', Locked = true;

    /// <summary>
    /// Ensures the Sales &amp; Receivables Setup is populated with the specified margin model.
    /// Deterministic: always leaves the setup in a known state for the test.
    /// </summary>
    procedure SetMarginModel(MarginModel: Enum "TSRD Margin Model")
    var
        SalesReceivablesSetup: Record "Sales & Receivables Setup";
    begin
        if not SalesReceivablesSetup.Get() then begin
            SalesReceivablesSetup.Init();
            SalesReceivablesSetup.Insert();
        end;
        SalesReceivablesSetup."TSRD Margin Model" := MarginModel;
        SalesReceivablesSetup.Modify();
    end;

    /// <summary>
    /// Creates (or resets) a Calc. Forecast test record with a deterministic key.
    /// The Key (Suffix) lets several forecasts coexist in one test.
    /// </summary>
    procedure CreateCalcForecast(Suffix: Code[5]) CalcForecast: Record "TSRD Calc. Forecast"
    var
        ForecastNo: Code[20];
    begin
        ForecastNo := CopyStr(StrSubstNo(CalcForecastNoTok, Suffix), 1, MaxStrLen(ForecastNo));
        if CalcForecast.Get(ForecastNo) then
            CalcForecast.Delete(true);
        Clear(CalcForecast);
        CalcForecast.Init();
        CalcForecast."No." := ForecastNo;
        CalcForecast.Description := 'F29748 Test Forecast';
        CalcForecast."Fix Price allowed" := false;
        CalcForecast.Insert();
    end;

    /// <summary>
    /// Creates or gets an Item Charge record usable as "Sales Item Charge" on the Cost Type
    /// (Calc. Type filter: Cost of Sale | Sales Freight).
    /// </summary>
    procedure GetOrCreateSalesItemCharge(Suffix: Code[5]): Code[20]
    var
        ItemCharge: Record "Item Charge";
        ItemChargeNo: Code[20];
    begin
        ItemChargeNo := CopyStr(StrSubstNo(ItemChargeCodeTok, Suffix), 1, MaxStrLen(ItemChargeNo));
        if not ItemCharge.Get(ItemChargeNo) then begin
            ItemCharge.Init();
            ItemCharge."No." := ItemChargeNo;
            ItemCharge.Description := 'F29748 Test Item Charge';
            ItemCharge."TSRM Calc. Type" := ItemCharge."TSRM Calc. Type"::"Cost of Sale";
            if ItemCharge.Insert() then;
        end;
        exit(ItemCharge."No.");
    end;

    /// <summary>
    /// Creates a Calc. Cost Type with a Sales Item Charge assigned,
    /// satisfying the validation of Charging Method = "Calc. Separately" on Calc. Cost.
    /// </summary>
    procedure CreateCostTypeWithSalesItemCharge(Suffix: Code[5]) CalcCostType: Record "TSRD Calc. Cost Type"
    var
        CostTypeCode: Code[20];
    begin
        CostTypeCode := CopyStr(StrSubstNo(CostTypeWithChargeCodeTok, Suffix), 1, MaxStrLen(CostTypeCode));
        if CalcCostType.Get(CostTypeCode) then
            CalcCostType.Delete();
        Clear(CalcCostType);
        CalcCostType.Init();
        CalcCostType."Code" := CostTypeCode;
        CalcCostType.Description := 'F29748 Cost Type with Charge';
        CalcCostType."Sales Item Charge" := GetOrCreateSalesItemCharge(Suffix);
        CalcCostType.Insert();
    end;

    /// <summary>
    /// Creates a Calc. Cost Type WITHOUT a Sales Item Charge, used for negative tests
    /// (Charging Method = "Calc. Separately" must fail).
    /// </summary>
    procedure CreateCostTypeWithoutSalesItemCharge(Suffix: Code[5]) CalcCostType: Record "TSRD Calc. Cost Type"
    var
        CostTypeCode: Code[20];
    begin
        CostTypeCode := CopyStr(StrSubstNo(CostTypeNoChargeCodeTok, Suffix), 1, MaxStrLen(CostTypeCode));
        if CalcCostType.Get(CostTypeCode) then
            CalcCostType.Delete();
        Clear(CalcCostType);

        CalcExtEqt."Line Amount (Direct Unit Cost)" := LineAmountCost;
        CalcExtEqt.Insert();
    end;

    /// <summary>

    /// test codeunit to guarantee determinism independent of test order.

    procedure ClearTestData()

        CalcWorkshop.DeleteAll(false);
        CalcExtEqt.SetFilter("Forecast No.", 'TST-F29748-*');

/// <summary>
/// Tests for User Story #31606 - "Extend Factbox to better represent separately
/// calculated Lines".
///
/// The acceptance criteria describe six new aggregation values that drive the
/// reworked Calc. Forecast FactBox:
///   - Additional Config. (Sep.) SP / Cost  (sum over Calc. External Equipment)
///   - Services (Separate) SP / Amount      (sum over Calc. Workshop)
///   - Costs (Separate) SP / Amount         (sum over Calc. Cost)
/// Each aggregation is filtered on Charging Method = Calc. Separately.
///
/// The tests below assert that exact aggregation rule by summing the underlying
/// records directly via SetRange + CalcSums – which is the data definition the
/// FactBox FlowFields encode. This keeps the tests stable against renames of the
/// FlowField captions while still verifying the acceptance criterion end to end.
/// </summary>
codeunit 81505 "TSRD F29748 FactBox Tests"
{
    Subtype = Test;
    TestPermissions = Disabled;

    var
        Assert: Codeunit "Library Assert";
        Lib: Codeunit "TSRD F29748 Charg. Method Lib";

    local procedure Initialize()
    begin
        Lib.ClearTestData();
        Lib.SetMarginModel(Enum::"TSRD Margin Model"::Surcharge);
    end;

    local procedure SumExtEqtSeparateSP(ForecastNo: Code[20]): Decimal
    var
        CalcExtEqt: Record "TSRD Calc. External Equipment";
    begin
        CalcExtEqt.SetRange("Forecast No.", ForecastNo);
        CalcExtEqt.SetRange("Charging Method", CalcExtEqt."Charging Method"::"Calc. Separately");
